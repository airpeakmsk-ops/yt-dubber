class SegmentStatus(StrEnum):
    PENDING    = "pending"
    TRANSLATED = "translated"
    APPROVED   = "approved"
    SKIPPED    = "skipped"
    TTS_DONE   = "tts_done"
    ERROR      = "error"

@dataclass
class SubtitleSegment:
    index:         int
    start_sec:     float
    end_sec:       float
    duration_sec:  float
    jp_text:       str
    ru_text:       str = ""
    audio_path:    Optional[str] = None
    status:        SegmentStatus = SegmentStatus.PENDING
    error_message: Optional[str] = None

@dataclass
class JobState:
    video_id:    str
    video_title: str
    source_url:  str
    docx_path:   str
    output_dir:  str
    segments:    list[SubtitleSegment] = field(default_factory=list)

    created_at:  str = ""
    updated_at:  str = ""
