    ```python
    if "youtube.com" not in url and "youtu.be" not in url:
        click.echo("Error: Not a YouTube URL. Expected youtube.com or youtu.be link.", err=True)
        sys.exit(1)
    if not settings.anthropic_api_key:
        click.echo("Error: ANTHROPIC_API_KEY is not set. Add it to .env and try again.", err=True)

    try:

        now = datetime.datetime.now().isoformat()

        checkpoint_path = docx_path.replace(".docx", "_checkpoint.json")
        checkpoint.save(job, checkpoint_path)
        click.echo(f"✅ Done! Review DOCX: {docx_path}")
        click.echo(f"   Then run: yt-dubber dub {docx_path}")
    ```

    dub() body (replacing raise NotImplementedError):
    ```python
    if not settings.elevenlabs_api_key:
        click.echo("Error: ELEVENLABS_API_KEY is not set. Add it to .env and try again.", err=True)
        sys.exit(1)

    checkpoint_path = docx_file.replace(".docx", "_checkpoint.json")
    if not os.path.exists(checkpoint_path):
        stem = os.path.splitext(os.path.basename(docx_file))[0]
        click.echo(
            f"Error: checkpoint not found for this DOCX. "
            f"Run 'yt-dubber translate <url>' first, or place "
            f"{stem}_checkpoint.json next to the DOCX.",
            err=True,
        )
        sys.exit(1)

    try:
        job = checkpoint.load(checkpoint_path)
        docx_edits = read_review_docx(docx_file)

        # Merge DOCX edits — DOCX is source of truth
        for seg in job.segments:
            if seg.index in docx_edits:
                seg.ru_text = docx_edits[seg.index]
            else:
                seg.status = SegmentStatus.SKIPPED
                seg.ru_text = ""

        # min_duration filter
        for seg in job.segments:
            if seg.status != SegmentStatus.SKIPPED and seg.duration_sec < min_duration:
                seg.status = SegmentStatus.SKIPPED
                seg.ru_text = ""

        effective_voice = voice_id or settings.elevenlabs_voice_id
        job.segments = tts.synthesize_all(
            job.segments, effective_voice, job.output_dir,
            resume=resume, job=job, checkpoint_path=checkpoint_path,
        )

        mp3_path = merger.assemble_track(
            job.segments,
            output_path=None,
            fmt=fmt,
            stretch=not no_stretch,
            job=job,
        )

        ok_count = sum(1 for s in job.segments if s.status == SegmentStatus.TTS_DONE)
        skipped_count = sum(1 for s in job.segments if s.status == SegmentStatus.SKIPPED)
        error_count = sum(1 for s in job.segments if s.status == SegmentStatus.ERROR)

        if error_count > 0:
            click.echo(f"WARN: {error_count} segments failed TTS (silence inserted)")

        click.echo(f"✅ Done! MP3: {mp3_path}")
        click.echo(f"   TTS: {ok_count} ok, {skipped_count} skipped, {error_count} errors")

    except Exception as exc:
        click.echo(f"Error: {exc}", err=True)
        sys.exit(1)
    ```
